       Console.WriteLine("So ngau nhien trong doan [15,80] la");

                if(SoNguyenTo(randomNumber)) 
                {
                    count=count+1;
                }
                 if(SoChinhPhuong(randomNumber)) 
                {
                    count1=count1+1;

            }
            Console.WriteLine("So luong so nguyen to trong doan tren la: {0}",count);

  }
  
  static bool SoNguyenTo(int n)
  {
    if (n < 2)
        return false;
    for (int i = 2; i < n; i++)
        if (n % i == 0)
            return false;
  
    return true;
  }
  static bool SoChinhPhuong(int n)
  {
            double sqrt = (int) Math.Sqrt(n);

            if(sqrt * sqrt == n){ return true;}
            else {return false; }  
  }
